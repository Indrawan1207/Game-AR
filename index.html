import cv2
import mediapipe as mp
import numpy as np
import time
import random
import math
from collections import deque
import threading

# =========================
# ADVANCED SOUND ENGINE
# =========================
try:
    import winsound
    def play_sound(kind):
        def _play():
            try:
                if kind == "slice": winsound.Beep(880, 35)
                elif kind == "bomb": winsound.Beep(200, 180)
                elif kind == "combo": winsound.Beep(1200, 45)
                elif kind == "start":
                    for f in [440, 880, 1320]:
                        winsound.Beep(f, 55)
            except: pass
        threading.Thread(target=_play, daemon=True).start()
except:
    def play_sound(kind): pass

# =========================
# AETHER ENGINE CORE
# =========================
class AetherEngine:
    def __init__(self):
        # MediaPipe lightweight configuration
        self.mp_hands = mp.solutions.hands
        self.hands = self.mp_hands.Hands(
            max_num_hands=2,
            model_complexity=0,
            min_detection_confidence=0.6,
            min_tracking_confidence=0.6
        )
        self.score = {'p1': 0, 'p2': 0}
        self.state = "MENU" # MENU, PLAY, GAMEOVER

    def run(self):
        # Inisialisasi Kamera 
        cap = cv2.VideoCapture(0)
        print("Engine Aether Blades Aktif...")
        
        while cap.isOpened():
            ret, frame = cap.read()
            if not ret: break
            
            # Balik Frame (Mirror)
            frame = cv2.flip(frame, 1)
            
            # --- Tampilkan UI dasar ---
            cv2.putText(frame, "AETHER BLADES PRO", (50, 50), 
                        cv2.FONT_HERSHEY_DUPLEX, 1, (255, 255, 255), 2)
            
            cv2.imshow("Aether Blades Showcase", frame)
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
                
        cap.release()
        cv2.destroyAllWindows()

if __name__ == "__main__":
    game = AetherEngine()
    game.run()
